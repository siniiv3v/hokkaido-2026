<script>
    const { createApp, ref } = Vue;

    createApp({
        setup() {
            const activeTab = ref('home');
            const commonFund = ref(50000);
            const expense = ref({ title: '', amount: null, payer: 'Common' });
            
            const schedule = ref([
                { time: '09:00', location: 'New Chitose Airport', desc: '' },
                { time: '13:00', location: 'Sapporo Station', desc: '' }
            ]);

            const shoppingList = ref([
                { name: 'LeTAO Cheesecake', price: '¬•2,500', img: null },
                { name: 'Shiroi Koibito', price: '¬•1,200', img: null }
            ]);

            const openMap = (location) => {
                const url = `https://www.google.com/maps/search/?api=1&query=${encodeURIComponent(location + ' Hokkaido')}`;
                window.open(url, '_blank');
            };

            const addExpense = () => {
                if (!expense.value.amount || !expense.value.title) return;
                if (expense.value.payer === 'Common') {
                    commonFund.value -= expense.value.amount;
                }
                alert(`Saved: ${expense.value.title}`);
                expense.value = { title: '', amount: null, payer: 'Common' };
            };

            const triggerPhotoUpload = () => {
                document.getElementById('photoInput').click();
            };

            const handlePhotoUpload = (event) => {
                const file = event.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = (e) => {
                        const name = prompt("Item Name?") || "Unknown";
                        shoppingList.value.push({
                            name: name,
                            price: "Pending",
                            img: e.target.result
                        });
                    };
                    reader.readAsDataURL(file);
                }
            };

            const removeItem = (type, index) => {
                if (confirm('Delete this item?')) {
                    if (type === 'schedule') schedule.value.splice(index, 1);
                    if (type === 'shopping') shoppingList.value.splice(index, 1);
                }
            };

            const addItem = (type) => {
                if (type === 'schedule') {
                    const loc = prompt("Destination?");
                    const time = prompt("Time? (e.g. 14:00)");
                    if (loc && time) schedule.value.push({ time, location: loc });
                }
            };

            return {
                activeTab, schedule, shoppingList, commonFund, expense,
                openMap, addExpense, triggerPhotoUpload, handlePhotoUpload,
                removeItem, addItem
            };
        }
    }).mount('#app');
</script>
